      }

        });

        item.sname_node_id = 'SECONDARY_NAMENODE_' + firstHostComponent.HostRoles.host_name;
      }
    });
    // Map
    var finalJson = this.parseIt(item, finalConfig);
    finalJson.quick_links = [1, 2, 3, 4];

    return finalJson;
  },
