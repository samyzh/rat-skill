# 功能抽象

  • base基类; 抽象出 ABase, Base

  • request再封装，默认子系统前缀

    /*
 * @Author: samy
 * @email: yessz#foxmail.com
 * @time: 2020-09-23 20:13:03
 * @modAuthor: samy
 * @modTime: 2021-01-27 00:55:26
 * @desc: request再封装一层
 * @Copyright © 2015~2021 BDP FE
 */
import requestBdp from '@/bdpcloud/utils/request';
import { apiPrefix } from '@<%=proName%>/conf/constant';

/**
 * 在原有的bdp中request方法中，再封装一层；
 * 方便处理当前子系统的特殊业务请求；
 */
export default function request(url, opts = {}, otherOpts = {}) {
  return requestBdp(`${apiPrefix}/${url}`, opts, otherOpts);
}
